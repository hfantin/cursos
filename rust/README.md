### floating point operations format
> let difference = 0.1 + 0.2;
> println!("{0:.2?}", difference);
> println!("{1:.0$?}", 2, difference);
> println!("{:.*?}", 2, difference);
> println!("{1:.*?}", 2, difference);
> println!("{number:.prec$?}", prec = 2, > > > number = difference);
